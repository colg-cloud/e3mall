# 1. e3 系统地址

```ini
# 后台管理系统
manager_url=http://127.0.0.1:9001/manager/index
# 前台首页
content_url=http://127.0.0.1:9002/content/index.html
# 搜索系统
search_url=http://127.0.0.1:9003/search/index.html
# 商品详情
item_url=http://127.0.0.1:9004/item/{itemId}.html
# 单点登录系统
# 注册
sso_register_url=http://127.0.0.1:9005/sso/register
# 登录; redirect: 登录以后的回调地址
sso_login_url=http://127.0.0.1:9005/sso/login?redirect=
```
